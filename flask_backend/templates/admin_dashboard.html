Zoekopdrachten van Gebruiker
document.addEventListener('DOMContentLoaded', function() { fetch('/api/admin/users') .then(response => response.json()) .then(data => { const tableBody = document.getElementById('usersTableBody'); tableBody.innerHTML = ''; let totalUsers = data.length; let totalSearches = 0; let emails = []; let searches = []; data.forEach(user => { const row = document.createElement('tr'); row.innerHTML = ` ${user.email} ${user.trial_active ? 'Ja' : 'Nee'} ${user.trial_days_left} ${user.trial_searches_left} ${user.searches_done} ${user.last_login || 'Nooit'} ${user.subscription_status} `; tableBody.appendChild(row); emails.push(user.email); searches.push(user.searches_done); totalSearches += user.searches_done; }); document.getElementById('totalUsers').textContent = totalUsers; document.getElementById('totalSearches').textContent = totalSearches; loadChart(emails, searches); }) .catch(error => console.error('Fout bij laden gebruikers:', error)); }); function viewSearches(userId) { fetch(`/api/admin/users/${userId}/searches`) .then(response => response.json()) .then(searches => { const searchesList = document.getElementById('searchesList'); searchesList.innerHTML = ''; if (searches.length === 0) { const listItem = document.createElement('li'); listItem.className = 'list-group-item'; listItem.textContent = 'Geen zoekopdrachten gevonden.'; searchesList.appendChild(listItem); } else { searches.forEach(search => { const listItem = document.createElement('li'); listItem.className = 'list-group-item'; listItem.textContent = `${search.timestamp} - ${search.query_text}`; searchesList.appendChild(listItem); }); } const modal = new bootstrap.Modal(document.getElementById('searchesModal')); modal.show(); }) .catch(error => console.error('Fout bij ophalen zoekopdrachten:', error)); } function loadChart(emails, searches) { const ctx = document.getElementById('searchChart').getContext('2d'); new Chart(ctx, { type: 'bar', data: { labels: emails, datasets: [{ label: 'Totaal Zoekopdrachten', data: searches, backgroundColor: 'rgba(54, 162, 235, 0.7)', borderColor: 'rgba(54, 162, 235, 1)', borderWidth: 1 }] }, options: { scales: { y: { beginAtZero: true } } } }); } function filterUsers() { const filter = document.getElementById('searchInput').value.toLowerCase(); const rows = document.querySelectorAll('#usersTableBody tr'); rows.forEach(row => { const text = row.textContent.toLowerCase(); if (text.includes(filter)) { row.style.display = ''; } else { row.style.display = 'none'; } }); } function debounce(func, delay) { let timeout; return function() { clearTimeout(timeout); timeout = setTimeout(func, delay); }; } document.getElementById('searchInput').addEventListener('input', debounce(filterUsers, 300));
